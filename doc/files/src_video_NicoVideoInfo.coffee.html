###*
# ニコニコ動画APIの動画情報モデルクラス

class NicoVideoInfo extends EventEmitter2
    @_cache     : {}


    ###*
    # オブジェクトがNicoVideoInfoのインスタンスか検証します。

    ###
    @isInstance : (obj) ->
        return obj instanceof VideoInfo


    @defaults    :
        title           : null
        description     : null
        length          : null      # 秒数
        movieType       : null      # "flv", "mp4"
        thumbnail       : null
        isDeleted       : false
        count           :
            view            : -1
            comments        : -1
            mylist          : -1

        tags            : []        # {name:string, isCategory:boolean, isLocked:boolean}

            id              :  -1
            name            : null
            icon            : null  # URL

    ###*

    ###*
    # @class NicoVideoInfo
    # @constructor
    # @param {String}       movieId     動画ID
    # @param {NicoSession}  session     セッション
    ###
    constructor     : (movieId, session) ->
        # 指定された動画の動画情報インスタンスがキャッシュされていればそれを返す
        # キャッシュに対応する動画情報インスタンスがなければ、新規作成してキャッシュ
        # return VideoInfo._cache[movieId] if VideoInfo._cache[movieId]?

        @_id = movieId
        @_session = session
        @attributes = _.cloneDeep(NicoVideoInfo.defaults)

        super

    ###*
    # 動画が削除されているか調べます。
    # @return {Boolean}
    ###
    isDeleted       : ->
      return @get "isDeleted"


    ###*
    # 属性を取得します。
    # @param {String}       attr        属性名(Ex. "id", "title", "user.id")
    ###
    get             : (attr) ->
        attr = attr.split "."
        ptr = @attributes

        while name = attr.pop()
            ptr = ptr[name]

            unless _.isObject(ptr) and attr.length is 0
                ptr = null
                break

        return ptr


    ###*
    # 最新の情報を取得します。
    # @method fetch
    # @return {Promise}
    # - resolve : (void)
    # - reject : (message: String)
    ###
    fetch           : ->
        return new Promise (resolve, reject) =>
            return reject "Fetch failed. Movie id not specified." unless @id?
            # console.error "[VideoInfo] Fetch failed. Movie id not specified."


            # getThumbInfoの結果を取得
            request.get
                url     : sprintf NicoURL.Video.GET_VIDEO_INFO, @_id
                jar     : @_session.getCookieJar()
            , (err, res, body) =>
                if err?
                    # console.error "VideoInfo[id:%s]: Failed to fetch movie info.", self.id

                    if res.statusCode is 503
                        errMsg = "Nicovideo has in maintenance."
                    else
                        errMsg = err

                    reject errMsg
                    @emit "error", @, errMsg


                @attributes = @parse(body)

                resolve()
                @emit "sync", @
                return


    ###*
    # @private
    # @param {String}       getThumbInfoAPIから取得したXML
    # @return {Object}
    ###
    _parse           : (res) ->
        $res = cheerio.load res

        if $res(":root").attr("status") isnt "ok"
            errCode = $res "error code"
            errMsg = $res("error description").text()
            # console.error "VideoInfo[%s]: 動画情報の取得に失敗しました。 (%s)", @id, errMsg

            return isDeleted: (errCode is "DELETED")


        $resThumb = $res "thumb"

        # 動画の秒単位の長さを出しておく
        length = do (length) ->
            length = $resThumb.find("length").text().split(":")
            s = length.pop() | 0
            m = length.pop() | 0
            h = length.pop() | 0
            return s + (m * 60) + (h * 3600)

        val =
            id          : $resThumb.find("video_id").text()
            title       : $resThumb.find("title").text()
            description : $resThumb.find("description").text()
            length      : length    # 秒数

            movieType   : $resThumb.find("movie_type").text()# "flv"とか
            thumbnail   : $resThumb.find("thumbnail_url").text()
            isDeleted   : false

            count       :
                view        : $resThumb.find("view_counter").text() | 0
                comments    : $resThumb.find("comment_num").text() | 0
                mylist      : $resThumb.find("mylist_counter").text() | 0

            tags        : do ->
                tags = []

                _.each $resThumb.find("tags"), (tags) ->
                    $tags = cheerio(tags)
                    domain = $tags.attr("domain")

                    _.each $tags.find("tag"), (tag) ->
                        $t = cheerio(tag)

                        tags.push
                            name        : $t.text()
                            isCategory  : $t.attr("category") is "1"
                            isLocked    : $t.attr("lock") is "1"
                            domain      : domain

                return tags

                # _.map $resThumb.find("tags[domain='jp'] tag"), (tag) ->
                #     $t = cheerio(tag)
                #     return {
                #         name        : $t.text()
                #         isCategory  : $t.attr("category") is "1"
                #         isLocked    : $t.attr("lock") is "1"
                #         domain      : "jp"
                #     }

            user        :
                id          : $resThumb.find("user_id").text() | 0
                name        : $resThumb.find("user_nickname").text()
                icon        : $resThumb.find("user_icon_url").text()

        return val


    ###*
    # インスタンスを破棄します。
    ###
    dispose : ->
        DisposeHelper.offAllEvents @
        DisposeHelper.wrapAllMembers @
