NicoSession     = require "./NicoSession"
NicoLive        = require "./api/live/NicoLiveApi"
NicoMyList      = require "./api/mylist/NicoMyListApi"
NicoVideo       = require "./api/video/NicoVideoApi"

###*
# ニコニコ動画APIのハンドラクラス
# 通常、`NicoNico.login`メソッドを利用して初期化します。
#
# @class NicoNico
###
class NicoNico
    @Session    = NicoSession
    @Live       = NicoLive
    @MyList     = NicoMyList
    @Video      = NicoVideo

    ###*
    # ニコニコ動画へログインし、ハンドラを取得します。
    #
    # @static
    # @method login
    # @param {String}   user        ログインユーザーID
    # @param {String}   password    ログインパスワード
    # @return {Promise}
    ###
    @login = (user, password) ->
        new Promise (resolve, reject) ->
            nico = new Nico user, password
            nico.session._login()
                .then -> resolve nico
                .catch (err) -> reject err


    ###*
    # @private
    # @property _session
    # @type NicoSession
    ###
    _session     : null

    ###*
    # @private
    # @property _live
    # @type NicoLive
    ###
    _live        : null # Lazy initialize

    ###*
    # @private
    # @property _video
    # @type NicoVideo
    ###
    _video       : null # Lazy initialize

    ###*
    # @private
    # @property _mylist
    # @type NicoMyList
    ###
    _mylist      : null # Lazy initialize


    ###*
    # @property session
    # @type NicoSession
    ###

    ###*
    # @property live
    # @type NicoLive
    ###

    ###*
    # @property video
    # @type NicoVideo
    ###

    ###*
    # @property mylist
    # @type NicoMyList
    ###



    ###*
    # @class NicoNico
    # @constructor
    # @param {String}       user        ユーザーID
    # @param {String}       password    ログインパスワード
    ###
    constructor     : (user, password) ->
        @_session = new NicoSession user, password

        Object.defineProperties @,
            session :
                get     : -> @_session
            live    :
                get     : -> @_live ?= new NicoLive @_session
            video   :
                get     : -> @_video ?= new NicoVideo @_session
            mylist  :
                get     : -> @_mylist ?= new NicoMyList @_session


    ###*
    # インスタンスに紐付いているオブジェクトを破棄し、利用不能にします。
    # @method dispose
    ###
    dispose         : ->
        @_session?.dispose()
        @_live?.dispose()
        @_video?.dispose()
        @_mylist?.dispose()
        return


module.exports = NicoNico
